import math
import matplotlib.pyplot as plt

def orb_num(f,x0,n):
    orbita = [x0]
    for _ in range (n):
        x0 = f(x0)
        orbita.append(x0)
    return orbita

    orb_entera = orb_num(f,x0,n)
    return orb_entera[m:]

import numpy as np

def an_grafico(x0, f, a, b, n):
    # Calcular la rbita usando orb_num
    orbit = orb_num(f, x0, n)
    
    # Creamos el linspace para poder mostrar mejor la grfica
    x_vals = np.linspace(a, b, 500)
    y_vals = f(x_vals)

    # Pintamos la f(x) en azul y ponemos la recta x=y con lnea discontinua
    plt.plot(x_vals, y_vals, 'blue')
    plt.plot(x_vals, x_vals, 'k--')
    
    x_current = x0
    for y_current in orbit[1:]:
        # Pintamos la lnea vertical desde el punto inicio hasta la grfica f(x)
        plt.plot([x_current, x_current], [x_current, y_current], 'r', lw=1)
        # Pintamos la lnea horizontal desde f(x) hasta la recta x=y
        plt.plot([x_current, y_current], [y_current, y_current], 'r', lw=1)
        # Definimos el nuevo punto de partida con x=y (est contenido en la recta)
        x_current = y_current

    plt.xlabel('x')
    plt.ylabel('y')
    plt.show()

# Ejemplo: mapa logstico
def f(x):
    return 3.9 * x * (1 - x)

# Parmetros
x0 = 0.2     # Valor inicial
a, b = 0, 1  # Intervalo de graficacin
n = 5    # Nmero de iteraciones

# Generar el diagrama cobweb
an_grafico(x0, f, a, b, n)
